## Overview

LunchTable TCG provides a library of themed UI components built with React and Tailwind CSS.

## Buttons

### Primary Button

```tsx
<Button variant="primary">Play Card</Button>
```

```css
.btn-primary {
  background: linear-gradient(135deg, var(--ltcg-gold-dark), var(--ltcg-gold));
  color: var(--ltcg-bg-primary);
  font-family: var(--ltcg-font-heading);
  font-weight: 600;
  padding: 0.75rem 1.5rem;
  border: none;
  border-radius: 4px;
  text-transform: uppercase;
  letter-spacing: 0.05em;
  cursor: pointer;
  transition: all 0.2s ease;
}

.btn-primary:hover {
  background: linear-gradient(135deg, var(--ltcg-gold), var(--ltcg-gold-light));
  box-shadow: 0 0 20px var(--ltcg-gold-glow);
}

.btn-primary:active {
  transform: translateY(1px);
}

.btn-primary:disabled {
  background: var(--ltcg-bg-hover);
  color: var(--ltcg-text-muted);
  cursor: not-allowed;
}
```

## Cards

### Card Frame

```tsx
<CardFrame element="arcane" rarity="rare">
  <CardContent card={card} />
</CardFrame>
```

```css

  position: relative;
  width: 250px;
  aspect-ratio: 2.5 / 3.5;
  background: var(--ltcg-bg-elevated);
  border: 2px solid var(--element-color);
  border-radius: 12px;
  overflow: hidden;
  box-shadow:
    0 0 20px var(--element-glow),
    inset 0 0 30px rgba(0, 0, 0, 0.5);

.card-frame::before {
  content: "";
  position: absolute;
  inset: 4px;
  border: 1px solid rgba(212, 175, 55, 0.3);
  border-radius: 8px;
  pointer-events: none;
}
```

## Form Elements

### Input Field

```tsx
<Input
  label="Deck Name"
  placeholder="Enter deck name..."
  value={name}
  onChange={setName}
/>
```

```css

  width: 100%;
  padding: 0.75rem 1rem;
  background: var(--ltcg-bg-primary);
  border: 1px solid var(--ltcg-gold-dark);
  border-radius: 4px;
  color: var(--ltcg-text-primary);
  font-family: var(--ltcg-font-body);
  font-size: 1rem;
  transition: all 0.2s ease;
}

.input:focus {
  outline: none;
  border-color: var(--ltcg-gold);
  box-shadow: 0 0 0 2px var(--ltcg-gold-glow);
}

.input::placeholder {
  color: var(--ltcg-text-muted);
}
```

## Modals

### Dialog

```tsx
<Dialog open={isOpen} onClose={close}>
  <Dialog.Title>Confirm Action</Dialog.Title>
  <Dialog.Content>
    Are you sure you want to discard this card?
  </Dialog.Content>
  <Dialog.Actions>
    <Button variant="secondary" onClick={close}>Cancel</Button>
    <Button variant="primary" onClick={confirm}>Confirm</Button>
  </Dialog.Actions>
</Dialog>
```

```css
.dialog-overlay {
  position: fixed;
  inset: 0;
  background: rgba(0, 0, 0, 0.8);
  backdrop-filter: blur(4px);
  display: flex;
  align-items: center;
  justify-content: center;
  z-index: 50;
}

.dialog {
  background: var(--ltcg-bg-elevated);
  border: 1px solid var(--ltcg-gold-dark);
  border-radius: 8px;
  padding: 1.5rem;
  max-width: 400px;
  width: 90%;
  box-shadow:
    0 0 40px rgba(0, 0, 0, 0.5),
    0 0 20px var(--ltcg-gold-glow);
}

.dialog-title {
  font-family: var(--ltcg-font-heading);
  font-size: 1.25rem;
  color: var(--ltcg-gold);
  margin-bottom: 1rem;
}
```

## Animation Classes

### Glow Pulse

```css
.glow-pulse {
  animation: glow-pulse 2s ease-in-out infinite;
}

@keyframes glow-pulse {
  0%, 100% {
    box-shadow: 0 0 10px var(--ltcg-gold-glow);
  }
  50% {
    box-shadow: 0 0 25px var(--ltcg-gold-glow);
  }
}
```
